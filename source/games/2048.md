<script>
  let board = [];
  let score = 0;
  let highScore = localStorage.getItem('2048HighScore') || 0;
  let gameOver = false;
  let won = false;
  let continueAfterWin = false;
  
  document.getElementById('highScore').textContent = highScore;
  
  function initBoard() {
    board = [];
    for (let i = 0; i < 4; i++) {
      board[i] = [];
      for (let j = 0; j < 4; j++) {
        board[i][j] = 0;
      }
    }
  }
  
  function addRandomTile() {
    const emptyCells = [];
    for (let i = 0; i < 4; i++) {
      for (let j = 0; j < 4; j++) {
        if (board[i][j] === 0) {
          emptyCells.push({ r: i, c: j });
        }
      }
    }
    
    if (emptyCells.length > 0) {
      const randomCell = emptyCells[Math.floor(Math.random() * emptyCells.length)];
      board[randomCell.r][randomCell.c] = Math.random() < 0.9 ? 2 : 4;
    }
  }
  
  function renderBoard() {
    const gameBoard = document.getElementById('gameBoard');
    gameBoard.innerHTML = '';
    
    for (let i = 0; i < 4; i++) {
      for (let j = 0; j < 4; j++) {
        const tile = document.createElement('div');
        const value = board[i][j];
        tile.className = `tile tile-${value > 2048 ? 'super' : value}`;
        tile.textContent = value || '';
        gameBoard.appendChild(tile);
      }
    }
  }
  
  function slide(row) {
    let arr = row.filter(val => val);
    let missing = 4 - arr.length;
    let zeros = Array(missing).fill(0);
    return zeros.concat(arr);
  }
  
  function combine(row) {
    for (let i = 3; i > 0; i--) {
      if (row[i] === row[i - 1] && row[i] !== 0) {
        row[i] *= 2;
        score += row[i];
        row[i - 1] = 0;
        
        if (row[i] === 2048 && !continueAfterWin) {
          won = true;
        }
      }
    }
    return row;
  }
  
  function moveRight() {
    let moved = false;
    for (let i = 0; i < 4; i++) {
      let row = board[i].slice();
      row = slide(row);
      row = combine(row);
      row = slide(row);
      
      if (row.join(',') !== board[i].join(',')) {
        moved = true;
      }
      board[i] = row;
    }
    return moved;
  }
  
  function moveLeft() {
    let moved = false;
    for (let i = 0; i < 4; i++) {
      let row = board[i].slice().reverse();
      row = slide(row);
      row = combine(row);
      row = slide(row);
      row.reverse();
      
      if (row.join(',') !== board[i].join(',')) {
        moved = true;
      }
      board[i] = row;
    }
    return moved;
  }
  
  function moveUp() {
    let moved = false;
    for (let j = 0; j < 4; j++) {
      let col = [];
      for (let i = 0; i < 4; i++) {
        col.push(board[i][j]);
      }
      
      let original = col.join(',');
      col.reverse();
      col = slide(col);
      col = combine(col);
      col = slide(col);
      col.reverse();
      
      if (col.join(',') !== original) {
        moved = true;
      }
      
      for (let i = 0; i < 4; i++) {
        board[i][j] = col[i];
      }
    }
    return moved;
  }
  
  function moveDown() {
    let moved = false;
    for (let j = 0; j < 4; j++) {
      let col = [];
      for (let i = 0; i < 4; i++) {
        col.push(board[i][j]);
      }
      
      let original = col.join(',');
      col = slide(col);
      col = combine(col);
      col = slide(col);
      
      if (col.join(',') !== original) {
        moved = true;
      }
      
      for (let i = 0; i < 4; i++) {
        board[i][j] = col[i];
      }
    }
    return moved;
  }
  
  function checkGameOver() {
    for (let i = 0; i < 4; i++) {
      for (let j = 0; j < 4; j++) {
        if (board[i][j] === 0) {
          return false;
        }
        
        if (i < 3 && board[i][j] === board[i + 1][j]) {
          return false;
        }
        
        if (j < 3 && board[i][j] === board[i][j + 1]) {
          return false;
        }
      }
    }
    return true;
  }
  
  function updateScore() {
    document.getElementById('score').textContent = score;
    
    if (score > highScore) {
      highScore = score;
      document.getElementById('highScore').textContent = highScore;
      localStorage.setItem('2048HighScore', highScore);
    }
  }
  
  function newGame() {
    initBoard();
    score = 0;
    gameOver = false;
    won = false;
    continueAfterWin = false;
    
    document.getElementById('score').textContent = score;
    document.getElementById('gameOver').classList.add('hidden');
    document.getElementById('gameWin').classList.add('hidden');
    
    addRandomTile();
    addRandomTile();
    renderBoard();
  }
  
  function continueGame() {
    continueAfterWin = true;
    document.getElementById('gameWin').classList.add('hidden');
  }
  
  function handleMove(direction) {
    if (gameOver) return;
    
    let moved = false;
    
    switch(direction) {
      case 'up':
        moved = moveUp();
        break;
      case 'down':
        moved = moveDown();
        break;
      case 'left':
        moved = moveLeft();
        break;
      case 'right':
        moved = moveRight();
        break;
    }
    
    if (moved) {
      addRandomTile();
      renderBoard();
      updateScore();
      
      if (won && !continueAfterWin) {
        document.getElementById('gameWin').classList.remove('hidden');
      } else if (checkGameOver()) {
        gameOver = true;
        document.getElementById('finalScore').textContent = score;
        document.getElementById('gameOver').classList.remove('hidden');
      }
    }
  }
  
  document.addEventListener('keydown', (e) => {
    if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key)) {
      e.preventDefault();
    }
    
    switch(e.key) {
      case 'ArrowUp':
      case 'w':
      case 'W':
        handleMove('up');
        break;
      case 'ArrowDown':
      case 's':
      case 'S':
        handleMove('down');
        break;
      case 'ArrowLeft':
      case 'a':
      case 'A':
        handleMove('left');
        break;
      case 'ArrowRight':
      case 'd':
      case 'D':
        handleMove('right');
        break;
    }
  });
  
  // 触摸支持
  let touchStartX = 0;
  let touchStartY = 0;
  
  document.addEventListener('touchstart', (e) => {
    touchStartX = e.touches[0].clientX;
    touchStartY = e.touches[0].clientY;
  });
  
  document.addEventListener('touchend', (e) => {
    const touchEndX = e.changedTouches[0].clientX;
    const touchEndY = e.changedTouches[0].clientY;
    
    const dx = touchEndX - touchStartX;
    const dy = touchEndY - touchStartY;
    
    if (Math.abs(dx) > Math.abs(dy)) {
      if (dx > 50) handleMove('right');
      else if (dx < -50) handleMove('left');
    } else {
      if (dy > 50) handleMove('down');
      else if (dy < -50) handleMove('up');
    }
  });
  
  // 初始化游戏
  newGame();
</script>
